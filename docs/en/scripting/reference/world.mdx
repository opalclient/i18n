# World & Entity Proxies

These proxies handle world queries, player status, and entity searching.

## ESP Proxy
**Global Binding:** `esp`

Projects 3D world positions to 2D screen coordinates for custom overlays.

### Entity Projection

<Method
    name="getEntityBox2D"
    signature="Vector4d getEntityBox2D(LivingEntity entity, float tickDelta)"
    description="Projects an entity's bounding box to screen coordinates. Returns null if the entity is off-screen."
    params={[
        { name: "entity", type: "LivingEntity", desc: "Target entity." },
        { name: "tickDelta", type: "float", desc: "Partial tick from client.getTickDelta()." }
    ]}
    returns={{ type: "Vector4d", desc: "{x, y, z(width), w(height)} or null." }}
/>

<Method
    name="isEntityOnScreen"
    signature="boolean isEntityOnScreen(LivingEntity entity, float tickDelta)"
    description="Returns true if any part of the entity's bounding box is visible on screen."
    params={[
        { name: "entity", type: "LivingEntity", desc: "Target entity." },
        { name: "tickDelta", type: "float", desc: "Partial tick." }
    ]}
    returns={{ type: "boolean", desc: "True if on screen." }}
/>
