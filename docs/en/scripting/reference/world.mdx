## ESP Proxy
**Global Binding:** `esp`

Helper for projecting 3D world coordinates to 2D screen space for custom overlays.

### Entity Projection

<Method
    name="getEntityBox2D"
    signature="Vector4d getEntityBox2D(LivingEntity entity, float tickDelta)"
    description="Projects an entity's bounding box to screen coordinates."
    params={[
        { name: "entity", type: "LivingEntity", desc: "Target entity." },
        { name: "tickDelta", type: "float", desc: "Interpolation factor." }
    ]}
    returns={{ type: "Vector4d", desc: "{x, y, z(width), w(height)} or null." }}
/>

<Method
    name="isEntityOnScreen"
    signature="boolean isEntityOnScreen(LivingEntity entity, float tickDelta)"
    description="Returns true if the entity is visible on screen."
    params={[
        { name: "entity", type: "LivingEntity", desc: "Target entity." },
        { name: "tickDelta", type: "float", desc: "Interpolation factor." }
    ]}
    returns={{ type: "boolean", desc: "True if visible." }}
/>
